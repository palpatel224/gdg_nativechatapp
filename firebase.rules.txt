// Firestore rules
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    match /users/{userId} {
      allow read: if request.auth != null;
      allow write: if request.auth != null && request.auth.uid == userId;
    }
    // other collections (messages, rooms) should have rules verifying auth and membership
  }
}

// Storage rules
service firebase.storage {
  match /b/{bucket}/o {
    match /users/{userId}/{allPaths=**} {
      allow read; // profile pictures are public by default; change if needed
      allow write: if request.auth != null && request.auth.uid == userId;
    }
  }
}
